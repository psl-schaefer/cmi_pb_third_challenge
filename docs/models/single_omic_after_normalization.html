make_flextable <- function(x) {
  if (RENDER) {
    x %>%
      flextable() %>% 
      bg(., bg = "#333333", part = "all") %>%
      color(., color = "white", part = "all") %>%
      set_table_properties(., align = "left") %>%
      flextable_to_rmd(ft) %>%
      return()
  } else {
    return(x)
  }
}

    make_flextable(.)
  
  cat("\n\n")

    #modality <- names(experimental_predictors)[3]

    modality_data <- experimental_predictors[[modality]] %>%
